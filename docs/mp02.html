load_atus_data <- function(file_base = c("resp", "rost", "act")){
  if(!dir.exists(file.path("data", "mp02"))){
    dir.create(file.path("data", "mp02"), showWarnings=FALSE, recursive=TRUE)
  }
  
  file_base <- match.arg(file_base)
  
  file_name_out <- file.path("data", 
                             "mp02", 
                             glue("atus{file_base}_0324.dat"))
  
  if(!file.exists(file_name_out)){
    
    url_end <- glue("atus{file_base}-0324.zip")
    
    temp_zip <- tempfile(fileext=".zip")
    temp_dir <- tools::file_path_sans_ext(temp_zip)
    
    request("https://www.bls.gov") |>
      req_url_path("tus", "datafiles", url_end) |>
      req_headers(`User-Agent` = "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.15; rv:143.0) Gecko/20100101 Firefox/143.0") |>
      req_perform(temp_zip)
    
    unzip(temp_zip, exdir = temp_dir)
    
    temp_file <- file.path(temp_dir, basename(file_name_out))
    
    file.copy(temp_file, file_name_out)
  }
  
  file <- read_csv(file_name_out, 
                   show_col_types = FALSE)
  
  switch(file_base, 
         resp=file |> 
           mutate(survey_weight = TUFNWGTP, 
                  time_alone = TRTALONE,
                  marriage_status = case_match(TRSPPRES, 1 ~ "Spouse Present", 2 ~ "Unmarried Spouse Present", 3 ~ "None"),
                  employment = case_match(TUABSOT, 1 ~ "Yes", 2 ~ "No", 3 ~ "Retired",4 ~ "Disabled",5 ~ "Unable"),
                  school = case_match(TESCHLVL, 1 ~ "HS", 2 ~ "Uni"),
                  income = TRERNWA,
                  time_alone = TRTALONE,
                  occupation = TRDTOCC1,
                  month = TUMONTH,
                  survey_year = TUYEAR,
                  number_children = TRCHILDNUM),
         
         rost=file |>
           mutate(sex = case_match(TESEX, 1 ~ "M", 2 ~ "F"),
                  individual = TULINENO,
                  age = TEAGE),
         act =file |>
           mutate(activity_n_of_day = TUACTIVITY_N, 
                  time_spent_min = TUACTDUR24, 
                  start_time = TUSTARTTIM,
                  stop_time = TUSTOPTIME, 
                  level1_code = paste0(TRTIER1P, "0000"), 
                  level2_code = paste0(TRTIER2P, "00"),
                  level3_code = TRCODEP,
                  children = TR_03CC57,
                  location = TEWHERE) 
  ) |> 
    rename(participant_id = TUCASEID) |>
    select(matches("[:lower:]", 
                   ignore.case=FALSE))
}

load_atus_activities <- function(){
  if(!dir.exists(file.path("data", "mp02"))){
    dir.create(file.path("data", "mp02"), showWarnings=FALSE, recursive=TRUE)
  }
  
  dest_file <- file.path("data",
                         "mp02",
                         "atus_activity_codes.csv")
  
  if(!file.exists(dest_file)){
    download.file("https://michael-weylandt.com/STA9750/mini/atus_activity_codes.csv", 
                  quiet=TRUE, 
                  destfile=dest_file, 
                  mode="wb")
  }
  
  read_csv(dest_file, show_col_types = FALSE) 
}

#resp_data = participant id, time alone, survey year, survey weight

rost_data <- load_atus_data("rost")

#rost_data = participant id, sex (household members)

act_data <- load_atus_data("act")

# act_Data = participant id, activity n of day, time spent, start time, stop time, 
#level 1 code, level 2 code, level 3 code

act_code <- load_atus_activities()

#act_code = code level 1-3, task level 1-3

#have three tables for each level of code

codelevel1 <- act_code |>
  select(code_level1, task_level1)

codelevel1 <- codelevel1 |>
  distinct(code_level1, .keep_all = TRUE)

codelevel2 <- act_code |>
  select(code_level2, task_level2)

codelevel2 <- codelevel2 |>
  distinct(code_level2, .keep_all = TRUE)

codelevel3 <- act_code |>
  select(code_level3, task_level3)

codelevel3 <- codelevel3 |>
  distinct(code_level3, .keep_all = TRUE)

#join act_data with codes

act_data_joined <- act_data |>
  left_join(codelevel1, by = c("level1_code" = "code_level1")) |>
  left_join(codelevel2, by = c("level2_code" = "code_level2")) |>
  left_join(codelevel3, by = c("level3_code" = "code_level3"))

full_data <- resp_data |>
  full_join(act_data_joined, by = "participant_id")

full_data <- full_data |>
  mutate(participant_id = as.character(participant_id))

rost_data <- rost_data |>
  mutate(participant_id = as.character(participant_id))

full_data_rost <- full_data |>
  full_join(rost_data, by = "participant_id", relationship = "many-to-many")
