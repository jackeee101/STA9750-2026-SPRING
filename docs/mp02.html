load_atus_data <- function(file_base = c("resp", "rost", "act")){
  if(!dir.exists(file.path("data", "mp02"))){
    dir.create(file.path("data", "mp02"), showWarnings=FALSE, recursive=TRUE)
  }
  
  file_base <- match.arg(file_base)
  
  file_name_out <- file.path("data", 
                             "mp02", 
                             glue("atus{file_base}_0324.dat"))
  
  if(!file.exists(file_name_out)){
    
    url_end <- glue("atus{file_base}-0324.zip")
    
    temp_zip <- tempfile(fileext=".zip")
    temp_dir <- tools::file_path_sans_ext(temp_zip)
    
    request("https://www.bls.gov") |>
      req_url_path("tus", "datafiles", url_end) |>
      req_headers(`User-Agent` = "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.15; rv:143.0) Gecko/20100101 Firefox/143.0") |>
      req_perform(temp_zip)
    
    unzip(temp_zip, exdir = temp_dir)
    
    temp_file <- file.path(temp_dir, basename(file_name_out))
    
    file.copy(temp_file, file_name_out)
  }
  
  file <- read_csv(file_name_out, 
                   show_col_types = FALSE)
  
  switch(file_base, 
         resp=file |> 
           mutate(survey_weight = TUFNWGTP, 
                  time_alone = TRTALONE,
                  marriage_status = case_match(TRSPPRES, 1 ~ "Spouse Present", 2 ~ "Unmarried Spouse Present", 3 ~ "None"),
                  employment = case_match(TUABSOT, 1 ~ "Yes", 2 ~ "No", 3 ~ "Retired",4 ~ "Disabled",5 ~ "Unable"),
                  school = case_match(TESCHLVL, 1 ~ "HS", 2 ~ "Uni"),
                  income = TRERNWA,
                  time_alone = TRTALONE,
                  occupation = TRDTOCC1,
                  month = TUMONTH,
                  survey_year = TUYEAR,
                  number_children = TRCHILDNUM),
         
         rost=file |>
           mutate(sex = case_match(TESEX, 1 ~ "M", 2 ~ "F"),
                  individual = TULINENO,
                  age = TEAGE),
         act =file |>
           mutate(activity_n_of_day = TUACTIVITY_N, 
                  time_spent_min = TUACTDUR24, 
                  start_time = TUSTARTTIM,
                  stop_time = TUSTOPTIME, 
                  level1_code = paste0(TRTIER1P, "0000"), 
                  level2_code = paste0(TRTIER2P, "00"),
                  level3_code = TRCODEP,
                  children = TR_03CC57,
                  location = case_when(
    TEWHERE == 1  ~ "Respondent's home",
    TEWHERE == 2  ~ "Workplace",
    TEWHERE == 3  ~ "Someone else's home",
    TEWHERE == 4  ~ "Restaurant/bar",
    TEWHERE == 5  ~ "Place of worship",
    TEWHERE == 6  ~ "Grocery store",
    TEWHERE == 7  ~ "Other store/mall",
    TEWHERE == 8  ~ "School",
    TEWHERE == 9  ~ "Outdoors away from home",
    TEWHERE == 10 ~ "Library",
    TEWHERE == 11 ~ "Car/vehicle",
    TEWHERE == 12 ~ "Other transportation",
    TEWHERE == 13 ~ "Bank",
    TEWHERE == 14 ~ "Gym/fitness center",
    TEWHERE == 15 ~ "Hospital/medical",
    TEWHERE == 16 ~ "Hotel/motel",
    TEWHERE == 17 ~ "Other place",
    TEWHERE == 18 ~ "Traveling",
    TRUE ~ NA_character_
  )) 
  ) |> 
    rename(participant_id = TUCASEID) |>
    select(matches("[:lower:]", 
                   ignore.case=FALSE))
}

resp_data <- load_atus_data("resp")

#resp_data = participant id, time alone, survey year, survey weight

rost_data <- load_atus_data("rost")

#rost_data = participant id, sex (household members)

act_data <- load_atus_data("act")

# act_Data = participant id, activity n of day, time spent, start time, stop time, 
#level 1 code, level 2 code, level 3 code

act_code <- load_atus_activities()

#act_code = code level 1-3, task level 1-3

#have three tables for each level of code

codelevel1 <- act_code |>
  select(code_level1, task_level1)

codelevel1 <- codelevel1 |>
  distinct(code_level1, .keep_all = TRUE)

codelevel2 <- act_code |>
  select(code_level2, task_level2)

codelevel2 <- codelevel2 |>
  distinct(code_level2, .keep_all = TRUE)

codelevel3 <- act_code |>
  select(code_level3, task_level3)

codelevel3 <- codelevel3 |>
  distinct(code_level3, .keep_all = TRUE)

#join act_data with codes

act_data_joined <- act_data |>
  left_join(codelevel1, by = c("level1_code" = "code_level1")) |>
  left_join(codelevel2, by = c("level2_code" = "code_level2")) |>
  left_join(codelevel3, by = c("level3_code" = "code_level3"))

full_data <- resp_data |>
  full_join(act_data_joined, by = "participant_id")

full_data <- full_data |>
  mutate(participant_id = as.character(participant_id))

rost_data <- rost_data |>
  mutate(participant_id = as.character(participant_id))

full_data_rost <- full_data |>
  full_join(rost_data, by = "participant_id", relationship = "many-to-many")

I load the data into different datasets and sliced up the activity code data into three datasets for easy joining. I then joined all the datasets excluding rost into full_data and a dataset including rost into full_data_rost. I mutated the participant id into a character format so I am able to see the entire id.

Op-Ed

In order to research the typical time use of the target demographics for the new time-tracking app, I utilized various sets of data from ATUS in the range 2003-2024. The target of the app are people wanting to better themselves. They want to track their every day activities and compare them against their peers. As such, the target demographics are “Stay at home” dads, college undergraduates, and late twenties to early thirties (26-34) individuals with high income and no kids to care for. I will be identifying parameters such as what activities they spent the most time on and what locations they are usually at. I will be disregarding the obvious activities and locations such as sleeping, eating and respondent’s home unless it is relevant. I will also be identifying parameters for activities people utilizing this app will want to track, such as sleep, personal care time and time spent exercising.

The first set of target demographics are “Stay at home” dads. I define “Stay at home” dads as male respondents with more than 0 kids and is unemployed. The most popular activities (besides sleeping and eating) for “Stay at home” dads are watching television, taking classes for degrees, and playing games. The most population locations (other than the respondent’s home) is school, someone else’s home and in a vehicle. This shows that most “Stay at home” dads invest a substantial amount of time trying to better themselves through obtaining skills and certifications. This shows that they would be interested in the app as it could help them with time management, between taking classes, time on hobbies, and ect.

The second set of target demographics are undergraduate students. I define undergraduate students as respondents still in Uni. The most popular activities (besides sleeping and eating) are working, watching television, research, and homework. The third most popular activity is “Research or homework for class (for degree, certification, or licensure)” meaning that undergraduate students are another set of demographic that majority work towards bettering themselves. The most popular locations for students (other than the respondent’s home) is the workplace, someone else’s home and school. For students, time management is imperative and for those with challenging courses, many seek out studying apps that track their time studying. Apps that limits the access of other apps to enhance studying efficiency is also very popular. As such, I believe that a time tracking app that allows them to track the time they spend on hobbies and work hours would be useful. I believe that if we can also incorporate a function where it limits other app usuages, it would be not only popular amount undergraduate students, but stay at home dads as well.

The third set of target demographics that I chose are individuals with high income and no kids. The most popular activities (besides sleeping and eating) is work, watching television, and socializing. The most popular locations for these individuals (besides their home) is their workplace, someone else’s home and other transportation. Although they do not have school or studying for a certificate among their popular activities, I believe that this demographic of individuals would also find a time tracking app useful. As our time tracking app is not only for studying but time management. Since they do not have children, compared to their peers with kids, they have more freedom with their time. They are able to track their sleep, personal care, and time spent exercising to better their health.

With the chart below, I calculated the target level for time spent on activities for the three demographics. I think that a good goal for users would be to meet the average time spent on these activities. if they wish to up the activity or allocate time differently, they could do so, but these will be the default settings.
