state_to_state_migration_2024 <- function() {

  data_dir <- file.path("data", "mp03")
    
  dir_path <- "data/mp03"
  if (!dir.exists(dir_path)) {
    dir.create(dir_path, recursive = TRUE)
    message("Created directory: ", dir_path)
  } 

  file_path <- file.path(dir_path, "State_to_State_Migration_Table_2024_T13.xlsx")

  url <- "https://www2.census.gov/programs-surveys/demo/tables/geographic-mobility/2024/state-to-state-migration/State_to_State_Migration_Table_2024_T13.xlsx"

  if (!file.exists(file_path)) {
    download.file(url, destfile = file_path, mode = "wb")
  }

  raw_data <- read_excel(file_path, skip = 6)
  raw_data2 <- read_excel(file_path, sheet = 2, skip = 6)
  
  raw_data <- raw_data |>
    mutate(
      Estimate = ifelse(Estimate == "N", 0, Estimate),
      `Margin of error3 (±)` = ifelse(`Margin of error3 (±)` == "N", 0, `Margin of error3 (±)`)) |>
    rename(state_current = ...1,
           state_1y = ...2)

  cleaned_data2 <- raw_data2 |>
    select(...1,Estimate...4,Estimate...6) |>
    filter(!is.na(...1) & ...1 != "United States5")|>
    mutate(sum = Estimate...4 + Estimate...6)

  raw_data <- raw_data |>
    left_join(cleaned_data2, by = c("state_current" = "...1"))

  raw_data <- raw_data |>
    select(-`Margin of error3 (±)`, -Estimate...4, -Estimate...6)

  raw_data <- raw_data |>
    mutate(
      Estimate = if_else(Estimate == "X",
                         as.numeric(sum),
                         as.numeric(Estimate))) |>
    select(-sum)

  raw_data <- raw_data |>
    rename(population = Estimate) |>
    mutate (year_current = 2024,
            year_1y = 2023)
  
  cleaned_2024data <- raw_data
  
  return(cleaned_2024data)
}

cleaned_2024data <- state_to_state_migration_2024()

state_to_state_migration_2023 <- function() {
  
   dir_path <- "data/mp03"
  if (!dir.exists(dir_path)) {
    dir.create(dir_path, recursive = TRUE)
    message("Created directory: ", dir_path)
  } 

  file_path <- file.path(dir_path, "State_to_State_Migration_Table_2023_T13.xlsx")

  url1 <- "https://www2.census.gov/programs-surveys/demo/tables/geographic-mobility/2023/state-to-state-migration/State_to_State_Migration_Table_2023_T13.xlsx"

  if (!file.exists(file_path)) {
    download.file(url1, destfile = file_path, mode = "wb")
  }

  raw_data3 <- read_excel(file_path, skip = 6)

  same_state2023 <- raw_data3 |>
    filter(!is.na(...1),
           !str_detect(...1, "United Stat")) |>
    select(...1,...4,...6) |>
    mutate(
      ...2 = ...1,
      population = as.numeric(...4) + as.numeric(...6)) |>
    select(-...4, -...6)

  same_state2023 <- same_state2023 |>
    rename(to_state = ...1,
           from_state = ...2)

  diff_state2023 <- raw_data3 |>
    filter(!is.na(...1),
           !str_detect(...1, "United Stat")) |>
    select(...1,any_of(state.name)) |>
    rename(to_state = ...1) |>
    pivot_longer(
      cols = -to_state,
      names_to = "from_state",
      values_to = "population") |>
    filter(!str_detect(population, "N/A"))

  same_state2023 <- same_state2023 |>
    mutate(population = as.numeric(population))

  diff_state2023 <- diff_state2023 |>
    mutate(population = as.numeric(gsub(",", "", population)))

  cleaned_2023data <- bind_rows(same_state2023, diff_state2023)

  cleaned_2023data <- cleaned_2023data |>
    mutate (year_current = 2023,
            year_1y = 2022) |>
    rename(state_current = to_state,
           state_1y = from_state)

  return(cleaned_2023data)
}

cleaned_2023data <- state_to_state_migration_2023()

  metro_to_metro_migration <- function() {

  dir_path <- "data/mp03"
  if (!dir.exists(dir_path)) {
    dir.create(dir_path, recursive = TRUE)
  }
  
  file_path <- file.path(dir_path, "metro_migration_2023.json")
  
  url <- "https://api.census.gov/data/2023/acs/flows?get=MOVEDIN,GEOID1,GEOID2,MOVEDOUT,FULL1_NAME,FULL2_NAME,MOVEDNET&for=metropolitan%20statistical%20area/micropolitan%20statistical%20area:*"

  if (!file.exists(file_path)) {
    download.file(url, destfile = file_path, mode = "wb")
  }

  metro_data <- read_json(file_path, simplifyVector = TRUE)

  move_row_to_colnames <- function(X, row = 1){
    X <- as_tibble(X)
    Xrow <- as.character(X[row, ])
    X <- X[-row, ]
    colnames(X) <- Xrow
    readr::type_convert(X)
  }

  metro_to_metro <- move_row_to_colnames(metro_data)

  pull_state_from_metro <- function(metro_name){
    str_extract(metro_name, ".*, (\\S{2})[-[:alpha:]]* Metro Area", group=1)
  }

  example_metros <- c(
    "Los Angeles-Long Beach-Anaheim, CA Metro Area", 
    "Dallas-Fort Worth-Arlington, TX Metro Area", 
    "Riverside-San Bernardino-Ontario, CA Metro Area", 
    "Atlanta-Sandy Springs-Alpharetta, GA Metro Area",
    "Chicago-Naperville-Elgin, IL-IN-WI Metro Area", 
    "New York-Newark-Jersey City, NY-NJ-PA Metro Area", 
    "Outside Metro Area within U.S. or Puerto Rico", 
    "Houston-The Woodlands-Sugar Land, TX Metro Area", 
    "Washington-Arlington-Alexandria, DC-VA-MD-WV Metro Area", 
    "San Jose-Sunnyvale-Santa Clara, CA Metro Area"
  )

  data.frame(metro = example_metros) |> 
    mutate(metro_state = pull_state_from_metro(metro)) |> 
    gt(id="tbl_metro_states") |> 
    cols_label(metro="Metropolitan Statistical Area", 
               metro_state="Primary State")

  return(metro_to_metro)
}

metro_to_metro <- metro_to_metro_migration()

As a policy analyst working for the governor of New York State, I was tasked with analyzing the sources of migration in order to find opportunities to encourage more people from either other states or other countries to migrate to New York. I will be approaching this topic by finding the major states and metro areas people usually migrate from to New York, and the areas people from New York tend to migrate to in order to try to reduce out-migration.

The first step I took was to explore the major sources of migration into New York from other states and metro areas. Individuals migrating to New York tend to come from New Jersey, California, Florida, and Pennsylvania. Upon taking a closer look, the metro areas they tend to come from include Philadelphia, Washington, D.C., Boston, Los Angeles, and other areas outside the U.S.

The second step I took was to explore the major source of migration out of New York to other states and metro areas. Individuals migrating out of New York tend to go to Florida, New Jersey, Pennsylvania, California, and Texas. Whereas the metro areas they tend to go to include Philadelphia, Miami, Washington, D.C., Los Angeles, and Bridgeport.

From the data, it is clear that people who migrate out tend to come from the same areas that New Yorkers migrate to, and they exhibit the same characteristics. Most are major walkable cities that have a large, diverse population. Which means, in order to keep migration low and increase migration, we have to enhance the qualities that make New York enticing. I also suggest limiting the efforts to cities further away, as residents from Philadelphia and New Jersey will always traverse back and forth to New York and vice versa. So the cities best to advertise to would be Los Angeles and Miami.

In order to increase the number of congressional seats, for my marketing campaign, I aim to increase migration by at least 600k - 1 million. My campaign consists of three parts: a hook to reel more people in, a reason to make them stay, and a reason to encourage them to retire here in New York in order to appeal to people of all ages. The younger crowd tends to migrate to the city, and the number of activities that take place increases when there is a large, congested population. However, a population with a skewed age distribution is not sustainable.

For the hook to reel more people in, I suggest investing in pop-up events in collaboration with creators and IPs (intellectual properties) that are popular. This will draw people from other states to visit the limited-time event and fall in love with the city while they are here. Activities are also a great way to build a community of like-minded people. Individuals who are in a community and have bonds that tether them to the state are less likely to move away. I think investing in safety and housing would also encourage people to stay and retire instead of moving away. If a more efficient transportation system connects the city with upstate New York, Staten Island, or Long Island.
